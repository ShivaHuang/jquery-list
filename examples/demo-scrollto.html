This list has dynamically added buttons which scroll to the next heading when clicked.

// Add a live click handler for the button.
$('#container dt button').live('click',function(){
    switch( $(this).attr('name') ){

        case 'top' :
            $('#container dl').list('scrollTo',0,'slow','swing');
        break;

        case 'next':
            var index = $('#container dl').list('header')+1;
            $('#container dl').list('scrollTo',index,'fast','swing');
        break;
    }
});

// Initialise the list.
$('#container dl').list().bind('headingChange',function(){

	// Add the 'next' button to the fake header element.
	$('#container .-list-fakeheader').append('<button name="next">Next</button>');
});

$('#container dt:first').append('<button name="next">Next</button>');
$('#container dt:last').append('<button name="top">Top</button>');
